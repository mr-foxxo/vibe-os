compat/bin
type: utilitario userland
dependencias provaveis: libc basica, argv, stdio, fs
viabilidade: alta

compat/etc
type: configuracao e dados
dependencias provaveis: instalacao no VFS
viabilidade: media

compat/games
type: aplicativos userland
dependencias provaveis: libc, terminal, possivel tty/input
viabilidade: media

compat/gnu
type: codigo importado de terceiros
dependencias provaveis: toolchain, headers, libc
viabilidade: media

compat/include
type: headers de compatibilidade
dependencias provaveis: integracao com build e compat runtime
viabilidade: alta

compat/lib
type: bibliotecas
dependencias provaveis: libc, posix, linker
viabilidade: media

compat/libc
type: biblioteca base
dependencias provaveis: memory, string, stdio, errno
viabilidade: alta

compat/libexec
type: utilitarios auxiliares
dependencias provaveis: processo, fs, runtime externo
viabilidade: baixa

compat/metadata
type: metadados do port
dependencias provaveis: nenhuma
viabilidade: alta

compat/posix
type: camada posix
dependencias provaveis: open, read, write, close, lseek, stat, isatty
viabilidade: alta

compat/regress
type: testes
dependencias provaveis: harness de testes, processo
viabilidade: baixa

compat/sbin
type: utilitario administrativo
dependencias provaveis: sistema, privilegios, fs
viabilidade: baixa

compat/share
type: dados, docs, terminfo, dicionarios
dependencias provaveis: instalacao no VFS
viabilidade: alta

compat/src
type: runtime adaptado do sistema
dependencias provaveis: libc, posix, build compat
viabilidade: alta

compat/sys
type: codigo de kernel de referencia
dependencias provaveis: kernel OpenBSD
viabilidade: nao portar para o kernel VibeOS

compat/unistd
type: wrappers de chamadas de sistema
dependencias provaveis: shell, VFS, syscalls
viabilidade: alta

compat/usr.bin
type: utilitarios userland
dependencias provaveis: libc, stdio, fs, terminal
viabilidade: alta para utilitarios pequenos, media/baixa para complexos

compat/usr.sbin
type: utilitarios administrativos
dependencias provaveis: rede, processo, privilegios
viabilidade: baixa
